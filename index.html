Getting Started

Download the production version or the development version.

In your web page:

<script src="jquery.js"></script>
<script src="dist/sfupload.min.js"></script>
<script>
$(function() {

    $('#files').sfupload({
        url: 'upload',
        allowdrop: true,
        onstart: function() {
            console.log('onstart');
        },
        onsuccess: function() {
            console.log('onsucess');
        },
        onprogress: function(data) {
            console.log('onprogress');
            console.log(data);
        }
    });

})
</script>
